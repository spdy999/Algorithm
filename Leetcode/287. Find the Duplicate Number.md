class Solution:
    def findDuplicate(self, nums: List[int]) -> int:
        s = f = 0
        while True:
            s = nums[s] # s = s.next 1
            f = nums[nums[f]] # f = f.next.next 3
            sv = nums[s] # 3
            fv = nums[f] # 2
            # print('s:', s, 'f:', f, 'sv:', sv, 'fv:', fv)
            if sv == fv:
                return sv
