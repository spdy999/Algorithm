```python
class Solution:
    def minCostClimbingStairs(self, cost: List[int]) -> int:
        n = len(cost)
        last_ind = n - 1
        mem = {}

        if n <= 2:
            return min(cost)
        
        def rec(i: int, summ: int) -> int:
            if i in mem:
                return summ + mem[i]
            if i > last_ind:
                return summ
            if i == last_ind or i == last_ind - 1:
                return summ + cost[i]                                   
            else:
                mini = min(rec(i+1, summ + cost[i]), rec(i+2, summ + cost[i]))
                mem[i] = mini - summ
                return mini
            
        return min(rec(0, 0), rec(1, 0))
```
