## [[HashSet]]
- Time : `O(n)`
- Space : `O(n)`
```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        cur = head
        check = set()

        while cur: # O(n)
            if cur in check:
                return True
            check.add(cur)
            cur = cur.next
        return False
```

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def hasCycle(self, head: Optional[ListNode]) -> bool:
        # Time: `O(n)`
        # Space: `O(1)`

        s, f = head, head

        while f and f.next:
            s = s.next
            f = f.next.next
            if s == f:
                return True

        return False

```
