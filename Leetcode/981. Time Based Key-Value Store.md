class TimeMap:

    def __init__(self):
        self.dict = defaultdict(list)
        

    def set(self, key: str, value: str, timestamp: int) -> None:
        self.dict[key].append((value, timestamp))
        

    def get(self, key: str, timestamp: int) -> str:
        # print('key:', key, 'time:', timestamp)
        # print(str(self.dict))
        if key not in self.dict:
            return None
        
        vals = self.dict[key] # beware of exception

        # binary search (time sorted)
        il, ir = 0, len(vals) - 1
        while il < ir:
            imid = (il + ir) // 2
	
            l = vals[il][1]
            r = vals[ir][1]
            mid = vals[imid][1]
            if mid == timestamp:
                return vals[mid][0]
            elif mid < timestamp:
                l = mid + 1
                if l > len(self.dict):
                    return None
            elif mid > timestamp:
                r = mid - 1
                if r < 0:
                    return None
                if vals[r][1] < mid:
                    return vals[r][0]
