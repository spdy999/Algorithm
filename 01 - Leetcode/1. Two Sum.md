class Solution:
    # TC: O(n), 1 pass
    def twoSum(self, nums: List[int], target: int) -> List[int]:
        hashmap = dict()
        for i, n in enumerate(nums):  # O(n)
            diff = target - n
            if diff in hashmap:
                return [i, hashmap[diff]]
            hashmap[n] = i
