# 269. Alien Dictionary
---

```python
class Solution:
    def foreignDictionary(self, words: List[str]) -> str:
        visit = dict() # {c: bool} False=visited True= Path
        adj = { c: set() for w in words for c in w }

        for i in range(len(words) - 1):
            w1, w2 = words[i], words[i+1]
            minLen = min(len(w1), len(w2))
            print(w1, w2)
            if len(w1) > len(w2) and w1[:minLen] == w2[:minLen]:
                return ""
            for j in range(minLen):
                if w1[j] != w2[j]:
                    adj[w1[j]].add(w2[j])

        print(adj)
        def dfs(c: str) -> bool:
            if c in visit:
                return visit[c]
            visit[c] = False
            for nei in adj[c]:
                if dfs(nei):
                    return True
                
            visit[c] = True
            return

        res = []
        res.reverse()
        return "".join(res)
```
